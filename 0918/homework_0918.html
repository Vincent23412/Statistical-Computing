import matplotlib.pyplot as plt
import numpy as np

n = 1000
x = np.linspace(-3*np.pi, 3*np.pi, n)
f = np.sin(x) / x
g = np.sin(x**2) / x
h = np.sin(2*x)**2 / (x**2)

fig, ax = plt.subplots(1)

ax.plot(x, f, color="r", label="$sin(x) / x$")
ax.plot(x, g, color="g", label="$sin(x^2) / x$")
ax.plot(x, h, color="b", label="$sin^2(2x) / x^2$")
ax.grid(True)
ax.set_xlabel('X')
ax.set_ylabel('Y', rotation=0)
# ax.set_title(
#     '$f(x) = \\frac{\\sin(x)}{x}, g(x) = \\frac{\\sin(x^{2})}{x}, h(x) = \\frac{\\sin^{2}(2x)}{x^{2}}$')
ax.set_xlim(-3*np.pi, 3*np.pi)
plt.legend()
plt.show()

n = 2000
x = np.linspace(-50, 50, n)

    f = np.exp(alpha * x) / (np.exp(alpha * x) + 1)
    ax.plot(x, f,   label=f"alpha = {alpha}")
ax.grid(True)
